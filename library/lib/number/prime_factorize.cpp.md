#include "../template.cpp"

// O(sqrt(n))
map<int64_t, int> prime_factorize(int64_t n) {
    map<int64_t, int> ret;
    for (int64_t i = 2; i * i <= n; ++i) {
        while (n % i == 0) {
            ret[i]++;
            n /= i;
        }
    }
    if (n != 1) ret[n] = 1;
    return ret;
}
